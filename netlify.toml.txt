# Netlify configuration for Brainfox Academy
# This file tells Netlify how to serve your site

[build]
  publish = "."        # Serve from root folder

# Security & performance headers
[[headers]]
  for = "/*"
  [headers.values]
    X-Frame-Options = "DENY"
    X-Content-Type-Options = "nosniff"
    Referrer-Policy = "strict-origin-when-cross-origin"
    Cache-Control = "public, max-age=0, must-revalidate"

# Cache static assets longer
[[headers]]
  for = "/Images/*"
  [headers.values]
    Cache-Control = "public, max-age=31536000, immutable"

# Redirect /admin to the CMS (added in Step 4)
[[redirects]]
  from = "/admin"
  to = "/admin/index.html"
  status = 200
